34 /* WGS84椭球模型参数
35  NOTE:如果使用其他椭球模型需要修改椭球参数 */
36 const double WGS84_WIE = 7.2921151467E-5; /* 地球自转角速度*/
37 const double WGS84_F = 0.0033528106647474805; /* 扁率 */
38 const double WGS84_RA = 6378137.0000000000; /* 长半轴a */
39 const double WGS84_RB = 6356752.3142451793; /* 短半轴b */
40 const double WGS84_GM0 = 398600441800000.00; /* 地球引力常数 */
41 const double WGS84_E1 = 0.0066943799901413156; /* 第一偏心率平方 */
42 const double WGS84_E2 = 0.0067394967422764341; /* 第二偏心率平方 */

47  /* 正常重力计算 */
48  static double gravity(const Vector3d &blh) {
49 
50  double sin2 = sin(blh[0]);
51  sin2 *= sin2;
52 
53  return 9.7803267715 * (1 + 0.0052790414 * sin2 + 0.0000232718 * sin2 * sin2) +
54  blh[2] * (0.0000000043977311 * sin2 - 0.0000030876910891) + 0.0000000000007211 * blh[2] * blh[2];
55  }

57  /* 计算子午圈半径和卯酉圈半径 */
58  static Eigen::Vector2d meridianPrimeVerticalRadius(double lat) {
59  double tmp, sqrttmp;
60 
61  tmp = sin(lat);
62  tmp *= tmp;
63  tmp = 1 - WGS84_E1 * tmp;
64  sqrttmp = sqrt(tmp);
65 
66  return {WGS84_RA * (1 - WGS84_E1) / (sqrttmp * tmp), WGS84_RA / sqrttmp};
67  }
68 
69  /* 计算卯酉圈半径 */
70  static double RN(double lat) {
71  double sinlat = sin(lat);
72  return WGS84_RA / sqrt(1.0 - WGS84_E1 * sinlat * sinlat);
73  }

75  /* n系(导航坐标系)到e系(地心地固坐标系)转换矩阵 */
76  static Matrix3d cne(const Vector3d &blh) {
77  double coslon, sinlon, coslat, sinlat;
78 
79  sinlat = sin(blh[0]);
80  sinlon = sin(blh[1]);
81  coslat = cos(blh[0]);
82  coslon = cos(blh[1]);
83 
84  Matrix3d dcm;
85  dcm(0, 0) = -sinlat * coslon;
86  dcm(0, 1) = -sinlon;
87  dcm(0, 2) = -coslat * coslon;
88 
89  dcm(1, 0) = -sinlat * sinlon;
90  dcm(1, 1) = coslon;
91  dcm(1, 2) = -coslat * sinlon;
92 
93  dcm(2, 0) = coslat;
94  dcm(2, 1) = 0;
95  dcm(2, 2) = -sinlat;
96 
97  return dcm;
98  }

100  /* n系(导航坐标系)到e系(地心地固坐标系)转换四元数 */
101  static Quaterniond qne(const Vector3d &blh) {
102  Quaterniond quat;
103 
104  double coslon, sinlon, coslat, sinlat;
105 
106  coslon = cos(blh[1] * 0.5);
107  sinlon = sin(blh[1] * 0.5);
108  coslat = cos(-M_PI * 0.25 - blh[0] * 0.5);
109  sinlat = sin(-M_PI * 0.25 - blh[0] * 0.5);
110 
111  quat.w() = coslat * coslon;
112  quat.x() = -sinlat * sinlon;
113  quat.y() = sinlat * coslon;
114  quat.z() = coslat * sinlon;
115 
116  return quat;
117  }
118 
119  /* 从n系到e系转换四元数得到纬度和经度 */
120  static Vector3d blh(const Quaterniond &qne, double height) {
121  return {-2 * atan(qne.y() / qne.w()) - M_PI * 0.5, 2 * atan2(qne.z(), qne.w()), height};
122  }

124  /* 大地坐标(纬度、经度和高程)转地心地固坐标 */
125  static Vector3d blh2ecef(const Vector3d &blh) {
126  double coslat, sinlat, coslon, sinlon;
127  double rnh, rn;
128 
129  coslat = cos(blh[0]);
130  sinlat = sin(blh[0]);
131  coslon = cos(blh[1]);
132  sinlon = sin(blh[1]);
133 
134  rn = RN(blh[0]);
135  rnh = rn + blh[2];
136 
137  return {rnh * coslat * coslon, rnh * coslat * sinlon, (rnh - rn * WGS84_E1) * sinlat};
138  }
139 
140  /* 地心地固坐标转大地坐标 */
141  static Vector3d ecef2blh(const Vector3d &ecef) {
142  double p = sqrt(ecef[0] * ecef[0] + ecef[1] * ecef[1]);
143  double rn;
144  double lat, lon;
145  double h = 0, h2;
146 
147  // 初始状态
148  lat = atan(ecef[2] / (p * (1.0 - WGS84_E1)));
149  lon = 2.0 * atan2(ecef[1], ecef[0] + p);
150 
151  do {
152  h2 = h;
153  rn = RN(lat);
154  h = p / cos(lat) - rn;
155  lat = atan(ecef[2] / (p * (1.0 - WGS84_E1 * rn / (rn + h))));
156  } while (fabs(h - h2) > 1.0e-4);
157 
158  return {lat, lon, h};
159  }

161  /* n系相对位置转大地坐标相对位置 */
162  static Matrix3d DRi(const Vector3d &blh) {
163  Matrix3d dri = Matrix3d::Zero();
164 
165  Eigen::Vector2d rmn = meridianPrimeVerticalRadius(blh[0]);
166 
167  dri(0, 0) = 1.0 / (rmn[0] + blh[2]);
168  dri(1, 1) = 1.0 / ((rmn[1] + blh[2]) * cos(blh[0]));
169  dri(2, 2) = -1;
170  return dri;
171  }
172 
173  /* 大地坐标相对位置转n系相对位置 */
174  static Matrix3d DR(const Vector3d &blh) {
175  Matrix3d dr = Matrix3d::Zero();
176 
177  Eigen::Vector2d rmn = meridianPrimeVerticalRadius(blh[0]);
178 
179  dr(0, 0) = rmn[0] + blh[2];
180  dr(1, 1) = (rmn[1] + blh[2]) * cos(blh[0]);
181  dr(2, 2) = -1;
182  return dr;
183  }

185  /* 局部坐标(在origin处展开)转大地坐标 */
186  static Vector3d local2global(const Vector3d &origin, const Vector3d &local) {
187 
188  Vector3d ecef0 = blh2ecef(origin);
189  Matrix3d cn0e = cne(origin);
190 
191  Vector3d ecef1 = ecef0 + cn0e * local;
192  Vector3d blh1 = ecef2blh(ecef1);
193 
194  return blh1;
195  }

197  /* 大地坐标转局部坐标(在origin处展开) */
198  static Vector3d global2local(const Vector3d &origin, const Vector3d &global) {
199  Vector3d ecef0 = blh2ecef(origin);
200  Matrix3d cn0e = cne(origin);
201 
202  Vector3d ecef1 = blh2ecef(global);
203 
204  return cn0e.transpose() * (ecef1 - ecef0);
205  }

207  static Pose local2global(const Vector3d &origin, const Pose &local) {
208  Pose global;
209 
210  Vector3d ecef0 = blh2ecef(origin);
211  Matrix3d cn0e = cne(origin);
212 
213  Vector3d ecef1 = ecef0 + cn0e * local.t;
214  Vector3d blh1 = ecef2blh(ecef1);
215  Matrix3d cn1e = cne(blh1);
216 
217  global.t = blh1;
218  global.R = cn1e.transpose() * cn0e * local.R;
219 
220  return global;
221  }

223  static Pose global2local(const Vector3d &origin, const Pose &global) {
224  Pose local;
225 
226  Vector3d ecef0 = blh2ecef(origin);
227  Matrix3d cn0e = cne(origin);
228 
229  Vector3d ecef1 = blh2ecef(global.t);
230  Matrix3d cn1e = cne(global.t);
231 
232  local.t = cn0e.transpose() * (ecef1 - ecef0);
233  local.R = cn0e.transpose() * cn1e * global.R;
234 
235  return local;

254  /* n系相对于e系转动角速度投影到n系 */
255  static Vector3d enwn(const Eigen::Vector2d &rmn, const Vector3d &blh, const Vector3d &vel) {
256  return {vel[1] / (rmn[1] + blh[2]), -vel[0] / (rmn[0] + blh[2]), -vel[1] * tan(blh[0]) / (rmn[1] + blh[2])};
257  }

259  static Vector3d enwn(const Vector3d &origin, const Vector3d &local, const Vector3d &vel) {
260  Vector3d global = local2global(origin, local);
261  Eigen::Vector2d rmn = meridianPrimeVerticalRadius(global[0]);
262 
263  return enwn(rmn, global, vel);
