78 class MicUtils
79 {
80 public:
81  /*
82  euler2dcm(roll, pitch, yaw)
83 
84  Converts a (Euler) roll-pitch-yaw (`X`-`Y`-`Z`) right-handed body to navigation
85  frame rotation (or the opposite rotation), to a DCM (direction cosine matrix).
86  Yaw is synonymous with azimuth and heading here.
87 
88  the body frame is rotated in the standard
89  -roll, -pitch, -yaw sequence to the navigation frame. For example, if v1 is
90  a 3x1 vector in the body frame [nose, right wing, down], then that vector
91  rotated into the navigation frame [north, east, down] would be v2 = dcm * v1.
92 
93  Reference: Titterton & Weston, Strapdown Inertial Navigation Technology, 2004,
94  Section 3.6 (pg. 36-41 & 537).
95 
96  **Arguments:**
97  - `roll`: length roll angle [rad], right-handed rotation about x-axis
98  - `pitch`: length pitch angle [rad], right-handed rotation about y-axis
99  - `yaw`: length yaw angle [rad], right-handed rotation about z-axis
100 
101  **Returns:**
102  - `dcm`: `3` x `3` direction cosine matrix [-]
103  */
104  static matrix_3f_t euler2dcm(float32_t roll, float32_t pitch, float32_t yaw)
105  {
106  float32_t cr = cos(roll);
107  float32_t sr = sin(roll);
108  float32_t cp = cos(pitch);
109  float32_t sp = sin(pitch);
110  float32_t cy = cos(yaw);
111  float32_t sy = sin(yaw);
112 
113  matrix_3f_t dcm = matrix_3f_t::Zero();
114  dcm(0, 0) = cp * cy;
115  dcm(0, 1) = -cr * sy + sr * sp * cy;
116  dcm(0, 2) = sr * sy + cr * sp * cy;
117  dcm(1, 0) = cp * sy;
118  dcm(1, 1) = cr * cy + sr * sp * sy;
119  dcm(1, 2) = -sr * cy + cr * sp * sy;
120  dcm(2, 0) = -sp;
121  dcm(2, 1) = sr * cp;
122  dcm(2, 2) = cr * cp;
123 
124  return dcm;
125  }
126 
127  static float32_t deg2rad(float32_t deg)
128  {
129  return deg*MIC_PI/180.0;
130  }
131 };
