44  // ZYX旋转顺序, 前右下的IMU, 输出RPY
45  static Vector3d matrix2euler(const Eigen::Matrix3d &dcm) {
46  Vector3d euler;
47 
48  euler[1] = atan(-dcm(2, 0) / sqrt(dcm(2, 1) * dcm(2, 1) + dcm(2, 2) * dcm(2, 2)));
49 
50  if (dcm(2, 0) <= -0.999) {
51  euler[0] = 0;
52  euler[2] = atan2((dcm(1, 2) - dcm(0, 1)), (dcm(0, 2) + dcm(1, 1)));
53  std::cout << "[WARNING] Rotation::matrix2euler: Singular Euler Angle! Set the roll angle to 0!" << std::endl;
54  } else if (dcm(2, 0) >= 0.999) {
55  euler[0] = 0;
56  euler[2] = M_PI + atan2((dcm(1, 2) + dcm(0, 1)), (dcm(0, 2) - dcm(1, 1)));
57  std::cout << "[WARNING] Rotation::matrix2euler: Singular Euler Angle! Set the roll angle to 0!" << std::endl;
58  } else {
59  euler[0] = atan2(dcm(2, 1), dcm(2, 2));
60  euler[2] = atan2(dcm(1, 0), dcm(0, 0));
61  }
62 
63  // heading 0~2PI
64  if (euler[2] < 0) {
65  euler[2] = M_PI * 2 + euler[2];
66  }
67 
68  return euler;
69  }
