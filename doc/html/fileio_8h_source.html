57 ret_t read_line(std::ifstream &infile, std::vector<float64_t> &data)
58 {
59  std::string line;

61  {
62  std::istringstream iss(line);
63  data.clear();
64  float64_t tmp;
65  while (iss >> tmp)
66  {
67  data.push_back(tmp);
68  }
69  return ret_t::MIC_RET_SUCCESSED;
70  }
71  return ret_t::MIC_RET_FAILED;

74 ret_t load_data(std::string file_name, mic_mag_compensator_shared_ptr mag_compensator_ptr)
75 {
76  std::ifstream infile(file_name);
77  if (!infile.is_open())
78  {
79  // MIC_LOG_ERR("failed open file %s", file_name);
80  return ret_t::MIC_RET_FAILED;
81  }
82 
83  while (true)
84  {
85  // float64_t ts, op_value, flux_x, flux_y, flux_z,
86  // op_truth, igrf_north, igrf_east, igrf_down,
87  // ins_pitch, ins_roll, ins_yaw;
88  // infile >> ts >> op_value >> flux_x >> flux_y >> flux_z >>
89  // op_truth >> igrf_north >> igrf_east >> igrf_down >>
90  // ins_pitch >> ins_roll >> ins_yaw;
91  // if (infile.eof())
92  // break;
93  std::vector<float64_t> data_line;
94  if (read_line(infile, data_line) == ret_t::MIC_RET_FAILED)
95  break;
96  float64_t ts = data_line[0];
97  float64_t op_value = data_line[1];
98  float64_t flux_x = data_line[2];
99  float64_t flux_y = data_line[3];
100  float64_t flux_z = data_line[4];
101  float64_t op_truth = data_line[5];
102  float64_t igrf_north = data_line[6];
103  float64_t igrf_east = data_line[7];
104  float64_t igrf_down = data_line[8];
105  float64_t ins_pitch = data_line[9];
106  float64_t ins_roll = data_line[10];
107  float64_t ins_yaw = data_line[11];
108 
109  mic_mag_t mag, mag_truth;
110  mic_nav_state_t nav_state;
111  nav_state.time_stamp = ts;
112  nav_state.attitude = quaternionf_t(
113  MicUtils::euler2dcm(
114  MicUtils::deg2rad(ins_roll),
115  MicUtils::deg2rad(ins_pitch),
116  MicUtils::deg2rad(ins_yaw)));
117  mag.time_stamp = ts;
118  mag.vector << flux_x, flux_y, flux_z;
119  mag.value = op_value;
120  mag_truth.time_stamp = ts;
121  mag_truth.vector << igrf_north, igrf_east, igrf_down;
122  mag_truth.value = op_truth;
123 
124  mag_compensator_ptr->add_data(ts, mag, nav_state);
125  mag_compensator_ptr->add_data_truth(ts, mag_truth); // TODO
126  }
127  infile.close();
128  return ret_t::MIC_RET_SUCCESSED;
129 }
