36 class CabinCostFunctionCreator
37 {
38 public:
39  CabinCostFunctionCreator(const matrix_3f_t &coeff_D_tilde_inv,
40  const vector_3f_t &coeff_o_hat,
41  const vector_3f_t &mag_r,
42  const vector_3f_t &mag_c)
43  : coeff_D_tilde_inv_(coeff_D_tilde_inv), coeff_o_hat_(coeff_o_hat),
44  mag_r_(mag_r), mag_c_(mag_c) {}
45 
46  template <typename T>
47  bool operator()(const T *const quaternion, T *residual_ptr) const
48  {
49  Eigen::Map<const Eigen::Quaternion<T>> q(quaternion);
50  Eigen::Matrix<T, 3, 3> R = q.toRotationMatrix();
51 
52  Eigen::Matrix<T, 3, 3> D_tilde_inv = coeff_D_tilde_inv_.template cast<T>();
53  Eigen::Matrix<T, 3, 1> o_hat = coeff_o_hat_.template cast<T>();
54  Eigen::Matrix<T, 3, 1> mag_r = mag_r_.template cast<T>();
55  Eigen::Matrix<T, 3, 1> mag_c = mag_c_.template cast<T>();
56 
57  Eigen::Matrix<T, 3, 1> mag_r1 = D_tilde_inv.inverse() * R * mag_c + o_hat;
58 
59  Eigen::Map<Eigen::Matrix<T, 3, 1>> residual(residual_ptr);
60  residual = mag_r - mag_r1;
61 
62  return true;
63  }

65  static ceres::CostFunction *Create(const matrix_3f_t &coeff_D_tilde_inv,
66  const vector_3f_t &coeff_o_hat,
67  const vector_3f_t &mag_r,
68  const vector_3f_t &mag_c)
69  {
70  return new ceres::AutoDiffCostFunction<CabinCostFunctionCreator, 3, 4>(
71  new CabinCostFunctionCreator(coeff_D_tilde_inv, coeff_o_hat,
72  mag_r, mag_c));
73  }
