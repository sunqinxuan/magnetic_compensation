40 // A compressed row sparse matrix used primarily for communicating the
41 // Jacobian matrix to the user.
42 struct CERES_EXPORT CRSMatrix {
43  CRSMatrix() : num_rows(0), num_cols(0) {}
44 
45  int num_rows;
46  int num_cols;
47 
48  // A compressed row matrix stores its contents in three arrays,
49  // rows, cols and values.
50  //
51  // rows is a num_rows + 1 sized array that points into the cols and
52  // values array. For each row i:
53  //
54  // cols[rows[i]] ... cols[rows[i + 1] - 1] are the indices of the
55  // non-zero columns of row i.
56  //
57  // values[rows[i]] .. values[rows[i + 1] - 1] are the values of the
58  // corresponding entries.
59  //
60  // cols and values contain as many entries as there are non-zeros in
61  // the matrix.
62  //
63  // e.g, consider the 3x4 sparse matrix
64  //
65  // [ 0 10 0 4 ]
66  // [ 0 2 -3 2 ]
67  // [ 1 2 0 0 ]
68  //
69  // The three arrays will be:
70  //
71  //
72  // -row0- ---row1--- -row2-
73  // rows = [ 0, 2, 5, 7]
74  // cols = [ 1, 3, 1, 2, 3, 0, 1]
75  // values = [10, 4, 2, -3, 2, 1, 2]
76 
77  std::vector<int> cols;
78  std::vector<int> rows;
79  std::vector<double> values;
80 };
