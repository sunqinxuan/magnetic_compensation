36 class CostFunctionCreator
37 {
38 public:
39  CostFunctionCreator(const matrix_3f_t &coeff_D_tilde_inv,
40  const vector_3f_t &coeff_o_hat,
41  const matrix_3f_t &rotation_r_c,
42  const vector_3f_t &mag_r,
43  const vector_3f_t &mag_c)
44  : coeff_D_tilde_inv_(coeff_D_tilde_inv), coeff_o_hat_(coeff_o_hat),
45  rotation_r_c_(rotation_r_c), mag_r_(mag_r), mag_c_(mag_c) {}
46 
47  template <typename T>
48  bool operator()(const T *const quaternion, T *residual_ptr) const
49  {
50  Eigen::Map<const Eigen::Quaternion<T>> q(quaternion);
51  Eigen::Matrix<T, 3, 3> R = q.toRotationMatrix();
52 
53  Eigen::Matrix<T, 3, 3> D_tilde_inv = coeff_D_tilde_inv_.template cast<T>();
54  Eigen::Matrix<T, 3, 1> o_hat = coeff_o_hat_.template cast<T>();
55  Eigen::Matrix<T, 3, 3> R_rc = rotation_r_c_.template cast<T>();
56  Eigen::Matrix<T, 3, 1> mag_r = mag_r_.template cast<T>();
57  Eigen::Matrix<T, 3, 1> mag_c = mag_c_.template cast<T>();
58 
59  Eigen::Matrix<T, 3, 1> mag_r1 = D_tilde_inv.inverse() * R * R_rc * R.transpose() * D_tilde_inv * (mag_c - o_hat) + o_hat;
60 
61  Eigen::Map<Eigen::Matrix<T, 3, 1>> residual(residual_ptr);
62  residual = mag_r - mag_r1;
63 
64  return true;
65  }
66 
67  static ceres::CostFunction *Create(const matrix_3f_t &coeff_D_tilde_inv,
68  const vector_3f_t &coeff_o_hat,
69  const matrix_3f_t &rotation_r_c,
70  const vector_3f_t &mag_r,
71  const vector_3f_t &mag_c)
72  {
73  return new ceres::AutoDiffCostFunction<CostFunctionCreator, 3, 4>(
74  new CostFunctionCreator(coeff_D_tilde_inv, coeff_o_hat, rotation_r_c,
75  mag_r, mag_c));
76  }
77 
78  EIGEN_MAKE_ALIGNED_OPERATOR_NEW
79 
80 private:
81  matrix_3f_t coeff_D_tilde_inv_;
82  vector_3f_t coeff_o_hat_;
83  matrix_3f_t rotation_r_c_; // rotation from b_k (c) to b_k+1 (r);
84  vector_3f_t mag_r_, mag_c_;
85 };

111  /* Ellipsoid fitting algorithm
112  *
113  * Inputs:
114  * x = nx1 column vector of x coordinates of input data.
115  * y = nx1 column vector of y coordinates of input data.
116  * z = nx1 column vector of y coordinates of input data.
117  *
118  * Output:
119  * u = [a,b,c,f,g,h,p,q,r,d], a vector corresponding to coefficients of
120  * the general quadric surface given by equation,
121  * ax2 + by2 + cz2 + 2fyz + 2gxz + 2hxy + 2px + 2qy + 2rz + d = 0.
122  *
123  * Source:
124  * [1] Li - Least Square Ellipsoid Fitting (2004)
125  * */
