65  // 用于输入数据的函数模板
66  template<typename T>
67  bool_t add_data(const float64_t ts, const T& data)
68  {
69  constexpr std::size_t index = tuple_index<T, std::tuple<Types...>>::value;
70  auto& data_map = std::get<index>(_data_storage);
71  data_map[ts] = data;
72  auto it = data_map.begin();
73  while (it != data_map.end() && it->first < ts - _data_remain_time)
74  {
75  it = data_map.erase(it);
76  }
77  return true;
78  }

80  // 用于获取数据的函数模板
81  template<typename T>
82  auto get_data_range(const float64_t start_ts, const float64_t end_ts)
83  {
84  constexpr std::size_t index = tuple_index<T, std::tuple<Types...>>::value;
85  auto& data_map = std::get<index>(_data_storage);
86  auto start_it = data_map.end();
87  auto end_it = data_map.end();
88  if (end_ts >= start_ts)
89  {
90  start_it = data_map.lower_bound(start_ts);
91  if (start_it != data_map.begin()) start_it--;
92  end_it = data_map.lower_bound(end_ts);
93  }
94  return std::make_pair(start_it, end_it);
95  }

118  // 递归处理每一个数据类型
119  template<typename T, typename... Rest>
120  bool_t get_data_impl(const float64_t ts, T& data, Rest&... rest)
121  {
122  constexpr std::size_t index = tuple_index<T, std::tuple<Types...> >::value;
123  auto& data_map = std::get<index>(_data_storage);
124  auto precise_iter = data_map.find(ts);
125  if (precise_iter != data_map.end())
126  {
127  data = precise_iter->second;
128  return get_data_impl(ts, rest...);
129  }
130  auto rough_iter_1 = data_map.lower_bound(ts);
131  auto rough_iter_2 = rough_iter_1;
132  if (rough_iter_2 != data_map.begin())
133  {
134  rough_iter_2--;
135  }
136  if (rough_iter_1 != data_map.end()
137  && fabs(ts - rough_iter_1->first) < _time_sync_tolerance)
138  {
139  data = rough_iter_1->second;
140  return get_data_impl(ts, rest...);
141  }
142  else if (rough_iter_2 != data_map.end()
143  && fabs(ts - rough_iter_2->first) < _time_sync_tolerance)
144  {
145  data = rough_iter_2->second;
146  return get_data_impl(ts, rest...);
147  }
148  return false;
149  }
150 
151  // 递归基例
152  bool_t get_data_impl(const float64_t ts)
153  {
154  return true;
155  }
