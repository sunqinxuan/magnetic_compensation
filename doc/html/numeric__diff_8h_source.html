92  // Mutates parameters but must restore them before return.
93  static bool EvaluateJacobianForParameterBlock(
94  const CostFunctor* functor,
95  const double* residuals_at_eval_point,
96  const NumericDiffOptions& options,
97  int num_residuals,
98  int parameter_block_index,
99  int parameter_block_size,
100  double **parameters,
101  double *jacobian) {
102  using Eigen::Map;
103  using Eigen::Matrix;
104  using Eigen::RowMajor;
105  using Eigen::ColMajor;
106 
107  const int num_residuals_internal =
108  (kNumResiduals != ceres::DYNAMIC ? kNumResiduals : num_residuals);
109  const int parameter_block_index_internal =
110  (kParameterBlock != ceres::DYNAMIC ? kParameterBlock :
111  parameter_block_index);
112  const int parameter_block_size_internal =
113  (kParameterBlockSize != ceres::DYNAMIC ? kParameterBlockSize :
114  parameter_block_size);
115 
116  typedef Matrix<double, kNumResiduals, 1> ResidualVector;
117  typedef Matrix<double, kParameterBlockSize, 1> ParameterVector;
118 
119  // The convoluted reasoning for choosing the Row/Column major
120  // ordering of the matrix is an artifact of the restrictions in
121  // Eigen that prevent it from creating RowMajor matrices with a
122  // single column. In these cases, we ask for a ColMajor matrix.
123  typedef Matrix<double,
124  kNumResiduals,
125  kParameterBlockSize,
126  (kParameterBlockSize == 1) ? ColMajor : RowMajor>
127  JacobianMatrix;
128 
129  Map<JacobianMatrix> parameter_jacobian(jacobian,
130  num_residuals_internal,
131  parameter_block_size_internal);
132 
133  Map<ParameterVector> x_plus_delta(
134  parameters[parameter_block_index_internal],
135  parameter_block_size_internal);
136  ParameterVector x(x_plus_delta);
137  ParameterVector step_size = x.array().abs() *
138  ((kMethod == RIDDERS) ? options.ridders_relative_initial_step_size :
139  options.relative_step_size);
140 
141  // It is not a good idea to make the step size arbitrarily
142  // small. This will lead to problems with round off and numerical
143  // instability when dividing by the step size. The general
144  // recommendation is to not go down below sqrt(epsilon).
145  double min_step_size = std::sqrt(std::numeric_limits<double>::epsilon());
146 
147  // For Ridders' method, the initial step size is required to be large,
148  // thus ridders_relative_initial_step_size is used.
149  if (kMethod == RIDDERS) {
150  min_step_size = std::max(min_step_size,
151  options.ridders_relative_initial_step_size);
152  }
153 
154  // For each parameter in the parameter block, use finite differences to
155  // compute the derivative for that parameter.
156  FixedArray<double> temp_residual_array(num_residuals_internal);
157  FixedArray<double> residual_array(num_residuals_internal);
158  Map<ResidualVector> residuals(residual_array.get(),
159  num_residuals_internal);
160 
161  for (int j = 0; j < parameter_block_size_internal; ++j) {
162  const double delta = std::max(min_step_size, step_size(j));
163 
164  if (kMethod == RIDDERS) {
165  if (!EvaluateRiddersJacobianColumn(functor, j, delta,
166  options,
167  num_residuals_internal,
168  parameter_block_size_internal,
169  x.data(),
170  residuals_at_eval_point,
171  parameters,
172  x_plus_delta.data(),
173  temp_residual_array.get(),
174  residual_array.get())) {
175  return false;
176  }
177  } else {
178  if (!EvaluateJacobianColumn(functor, j, delta,
179  num_residuals_internal,
180  parameter_block_size_internal,
181  x.data(),
182  residuals_at_eval_point,
183  parameters,
184  x_plus_delta.data(),
185  temp_residual_array.get(),
186  residual_array.get())) {
187  return false;
188  }
189  }
190 
191  parameter_jacobian.col(j).matrix() = residuals;
192  }
193  return true;
194  }

196  static bool EvaluateJacobianColumn(const CostFunctor* functor,
197  int parameter_index,
198  double delta,
199  int num_residuals,
200  int parameter_block_size,
201  const double* x_ptr,
202  const double* residuals_at_eval_point,
203  double** parameters,
204  double* x_plus_delta_ptr,
205  double* temp_residuals_ptr,
206  double* residuals_ptr) {
207  using Eigen::Map;
208  using Eigen::Matrix;
209 
210  typedef Matrix<double, kNumResiduals, 1> ResidualVector;
211  typedef Matrix<double, kParameterBlockSize, 1> ParameterVector;
212 
213  Map<const ParameterVector> x(x_ptr, parameter_block_size);
214  Map<ParameterVector> x_plus_delta(x_plus_delta_ptr,
215  parameter_block_size);
216 
217  Map<ResidualVector> residuals(residuals_ptr, num_residuals);
218  Map<ResidualVector> temp_residuals(temp_residuals_ptr, num_residuals);
219 
220  // Mutate 1 element at a time and then restore.
221  x_plus_delta(parameter_index) = x(parameter_index) + delta;
222 
223  if (!EvaluateImpl<CostFunctor, N0, N1, N2, N3, N4, N5, N6, N7, N8, N9>(
224  functor, parameters, residuals.data(), functor)) {
225  return false;
226  }
227 
228  // Compute this column of the jacobian in 3 steps:
229  // 1. Store residuals for the forward part.
230  // 2. Subtract residuals for the backward (or 0) part.
231  // 3. Divide out the run.
232  double one_over_delta = 1.0 / delta;
233  if (kMethod == CENTRAL || kMethod == RIDDERS) {
234  // Compute the function on the other side of x(parameter_index).
235  x_plus_delta(parameter_index) = x(parameter_index) - delta;
236 
237  if (!EvaluateImpl<CostFunctor, N0, N1, N2, N3, N4, N5, N6, N7, N8, N9>(
238  functor, parameters, temp_residuals.data(), functor)) {
239  return false;
240  }
241 
242  residuals -= temp_residuals;
243  one_over_delta /= 2;
244  } else {
245  // Forward difference only; reuse existing residuals evaluation.
246  residuals -=
247  Map<const ResidualVector>(residuals_at_eval_point,
248  num_residuals);
249  }
250 
251  // Restore x_plus_delta.
252  x_plus_delta(parameter_index) = x(parameter_index);
253 
254  // Divide out the run to get slope.
255  residuals *= one_over_delta;
256 
257  return true;
258  }

260  // This numeric difference implementation uses adaptive differentiation
261  // on the parameters to obtain the Jacobian matrix. The adaptive algorithm
262  // is based on Ridders' method for adaptive differentiation, which creates
263  // a Romberg tableau from varying step sizes and extrapolates the
264  // intermediate results to obtain the current computational error.
265  //
266  // References:
267  // C.J.F. Ridders, Accurate computation of F'(x) and F'(x) F"(x), Advances
268  // in Engineering Software (1978), Volume 4, Issue 2, April 1982,
269  // Pages 75-76, ISSN 0141-1195,
270  // http://dx.doi.org/10.1016/S0141-1195(82)80057-0.
271  static bool EvaluateRiddersJacobianColumn(
272  const CostFunctor* functor,
273  int parameter_index,
274  double delta,
275  const NumericDiffOptions& options,
276  int num_residuals,
277  int parameter_block_size,
278  const double* x_ptr,
279  const double* residuals_at_eval_point,
280  double** parameters,
281  double* x_plus_delta_ptr,
282  double* temp_residuals_ptr,
283  double* residuals_ptr) {
284  using Eigen::Map;
285  using Eigen::Matrix;
286  using Eigen::aligned_allocator;
287 
288  typedef Matrix<double, kNumResiduals, 1> ResidualVector;
289  typedef Matrix<double, kNumResiduals, Eigen::Dynamic> ResidualCandidateMatrix;
290  typedef Matrix<double, kParameterBlockSize, 1> ParameterVector;
291 
292  Map<const ParameterVector> x(x_ptr, parameter_block_size);
293  Map<ParameterVector> x_plus_delta(x_plus_delta_ptr,
294  parameter_block_size);
295 
296  Map<ResidualVector> residuals(residuals_ptr, num_residuals);
297  Map<ResidualVector> temp_residuals(temp_residuals_ptr, num_residuals);
298 
299  // In order for the algorithm to converge, the step size should be
300  // initialized to a value that is large enough to produce a significant
301  // change in the function.
302  // As the derivative is estimated, the step size decreases.
303  // By default, the step sizes are chosen so that the middle column
304  // of the Romberg tableau uses the input delta.
305  double current_step_size = delta *
306  pow(options.ridders_step_shrink_factor,
307  options.max_num_ridders_extrapolations / 2);
308 
309  // Double-buffering temporary differential candidate vectors
310  // from previous step size.
311  ResidualCandidateMatrix stepsize_candidates_a(
312  num_residuals,
313  options.max_num_ridders_extrapolations);
314  ResidualCandidateMatrix stepsize_candidates_b(
315  num_residuals,
316  options.max_num_ridders_extrapolations);
317  ResidualCandidateMatrix* current_candidates = &stepsize_candidates_a;
318  ResidualCandidateMatrix* previous_candidates = &stepsize_candidates_b;
319 
320  // Represents the computational error of the derivative. This variable is
321  // initially set to a large value, and is set to the difference between
322  // current and previous finite difference extrapolations.
323  // norm_error is supposed to decrease as the finite difference tableau
324  // generation progresses, serving both as an estimate for differentiation
325  // error and as a measure of differentiation numerical stability.
326  double norm_error = std::numeric_limits<double>::max();
327 
328  // Loop over decreasing step sizes until:
329  // 1. Error is smaller than a given value (ridders_epsilon),
330  // 2. Maximal order of extrapolation reached, or
331  // 3. Extrapolation becomes numerically unstable.
332  for (int i = 0; i < options.max_num_ridders_extrapolations; ++i) {
333  // Compute the numerical derivative at this step size.
334  if (!EvaluateJacobianColumn(functor, parameter_index, current_step_size,
335  num_residuals,
336  parameter_block_size,
337  x.data(),
338  residuals_at_eval_point,
339  parameters,
340  x_plus_delta.data(),
341  temp_residuals.data(),
342  current_candidates->col(0).data())) {
343  // Something went wrong; bail.
344  return false;
345  }
346 
347  // Store initial results.
348  if (i == 0) {
349  residuals = current_candidates->col(0);
350  }
351 
352  // Shrink differentiation step size.
353  current_step_size /= options.ridders_step_shrink_factor;
354 
355  // Extrapolation factor for Richardson acceleration method (see below).
356  double richardson_factor = options.ridders_step_shrink_factor *
357  options.ridders_step_shrink_factor;
358  for (int k = 1; k <= i; ++k) {
359  // Extrapolate the various orders of finite differences using
360  // the Richardson acceleration method.
361  current_candidates->col(k) =
362  (richardson_factor * current_candidates->col(k - 1) -
363  previous_candidates->col(k - 1)) / (richardson_factor - 1.0);
364 
365  richardson_factor *= options.ridders_step_shrink_factor *
366  options.ridders_step_shrink_factor;
367 
368  // Compute the difference between the previous value and the current.
369  double candidate_error = std::max(
370  (current_candidates->col(k) -
371  current_candidates->col(k - 1)).norm(),
372  (current_candidates->col(k) -
373  previous_candidates->col(k - 1)).norm());
374 
375  // If the error has decreased, update results.
376  if (candidate_error <= norm_error) {
377  norm_error = candidate_error;
378  residuals = current_candidates->col(k);
379 
380  // If the error is small enough, stop.
381  if (norm_error < options.ridders_epsilon) {
382  break;
383  }
384  }
385  }
386 
387  // After breaking out of the inner loop, declare convergence.
388  if (norm_error < options.ridders_epsilon) {
389  break;
390  }
391 
392  // Check to see if the current gradient estimate is numerically unstable.
393  // If so, bail out and return the last stable result.
394  if (i > 0) {
395  double tableau_error = (current_candidates->col(i) -
396  previous_candidates->col(i - 1)).norm();
397 
398  // Compare current error to the chosen candidate's error.
399  if (tableau_error >= 2 * norm_error) {
400  break;
401  }
402  }
403 
404  std::swap(current_candidates, previous_candidates);
405  }
406  return true;
407  }
