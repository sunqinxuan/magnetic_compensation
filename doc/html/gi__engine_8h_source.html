90  static void imuInterpolate(const IMU &imu1, IMU &imu2, const double timestamp, IMU &midimu) {
91 
92  if (imu1.time > timestamp || imu2.time < timestamp) {
93  return;
94  }
95 
96  double lamda = (timestamp - imu1.time) / (imu2.time - imu1.time);
97 
98  midimu.time = timestamp;
99  midimu.dtheta = imu2.dtheta * lamda;
100  midimu.dvel = imu2.dvel * lamda;
101  midimu.dt = timestamp - imu1.time;
102 
103  imu2.dtheta = imu2.dtheta - midimu.dtheta;
104  imu2.dvel = imu2.dvel - midimu.dvel;
105  imu2.dt = imu2.dt - midimu.dt;
106  }
